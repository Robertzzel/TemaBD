Pragma foreign_keys = ON
CREATE TABLE CATALOG(NR_MATRICOL INTEGER PRIMARY KEY, ULTIMA_MEDIE INTEGER)
CREATE TABLE FACULTATI(ID_FACULTATE INTEGER PRIMARY KEY, NR_PROFESORI INTEGER, NR_ELEVI INTEGER, LOCATIE TEXT)
CREATE TABLE PROFESORI(ID_PROFESOR INTEGER PRIMARY KEY, ID_FACULTATE INTEGER, NR_MATERII_PREDATE INTEGER, SALARIU REAL, FOREIGN KEY (ID_FACULTATE) REFERENCES FACULTATI(ID_FACULTATE))
CREATE TABLE ELEVI(NR_MATRICOL INTEGER PRIMARY KEY, NUME TEXT, PRENUME TEXT, ID_FACULTATE INTEGER, ID_INDRUMATOR INTEGER,FOREIGN KEY (ID_FACULTATE) REFERENCES FACULTATI(ID_FACULTATE), FOREIGN KEY (ID_INDRUMATOR) REFERENCES PROFESORI(ID_PROFESOR) , FOREIGN KEY (NR_MATRICOL) REFERENCES CATALOG(NR_MATRICOL))

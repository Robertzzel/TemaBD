CREATE TABLE CATALOAGE(NR_MATRICOL INTEGER PRIMARY KEY, ULTIMA_MEDIE REAL , CONSTRAINT my_Check CHECK(ULTIMA_MEDIE <= 10 AND ULTIMA_MEDIE >0))

CREATE TABLE PROFESORI(ID_PROFESOR INTEGER PRIMARY KEY, ID_FACULTATE INTEGER, NR_MATERII_PREDATE INTEGER, SALARIU REAL, FOREIGN KEY (ID_FACULTATE) REFERENCES FACULTATI(ID_FACULTATE))
CREATE TABLE ELEVI(NR_MATRICOL INTEGER PRIMARY KEY, NUME TEXT, PRENUME TEXT, ID_FACULTATE INTEGER, ID_INDRUMATOR INTEGER,FOREIGN KEY (ID_FACULTATE) REFERENCES FACULTATI(ID_FACULTATE) ON DELETE CASCADE, FOREIGN KEY (ID_INDRUMATOR) REFERENCES PROFESORI(ID_PROFESOR) , FOREIGN KEY (NR_MATRICOL) REFERENCES CATALOAGE(NR_MATRICOL) ON DELETE CASCADE)
